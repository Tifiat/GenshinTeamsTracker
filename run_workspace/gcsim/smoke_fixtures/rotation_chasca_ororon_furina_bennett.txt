energy every interval=480,720 amount=1;

  if is_even(i) {
    furina skill, dash, burst;
  }
  ororon burst, skill;
  bennett skill, burst;
  chasca skill;
  chasca aim:4;
  chasca aim[bullets=4];
  if is_even(i) {
    chasca burst;
  }
}
